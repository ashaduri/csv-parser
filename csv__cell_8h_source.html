373 template<std::size_t Size>
374 constexpr typename CellStringBuffer<Size>::Buffer CellStringBuffer<Size>::prepareBuffer(std::string_view input, bool
375 has_escaped_quotes)
376 {
377  if (Size < input.size()) {
378  return Buffer{};
379  }
380  if (has_escaped_quotes) {
381  return cleanString(input);
382  }
383  std::array<char, Size> buffer = { };
384  for (std::size_t pos = 0; pos < std::min(Size, input.size()); ++pos) {
385  buffer[pos] = input[pos];
386  }
387  return Buffer{buffer, input.size(), true};
388 }

392 template<std::size_t Size>
393 constexpr typename CellStringBuffer<Size>::Buffer CellStringBuffer<Size>::cleanString(std::string_view input)
394 {
395  std::array<char, Size> buffer = { };
396  std::size_t output_pos = 0;
397  for (std::size_t input_pos = 0; input_pos < std::min(Size, input.size()); ++input_pos) {
398  char c = input[input_pos];
399  buffer[output_pos] = c;
400  ++output_pos;
401  if (c == '\"' && (input_pos + 1) < input.size() && input[input_pos + 1] == '\"') {
402  ++input_pos;
403  }
404  }
405  return {buffer, output_pos, true};
406 }

646 template<typename CellT>
647 constexpr CellT CellTrait<CellT>::create(std::string_view cell, CellTypeHint hint)
648 {
649  // The types here are the same as in readNumber()
650  if constexpr(std::is_same_v<CellT, float>
651  || std::is_same_v<CellT, double>
652  || std::is_same_v<CellT, long double>) {
653  return readNumber<CellT>(cell).value_or(std::numeric_limits<CellT>::quiet_NaN());
654  } else if constexpr(std::is_same_v<CellT, int>
655  || std::is_same_v<CellT, long int>
656  || std::is_same_v<CellT, long long int>
657  || std::is_same_v<CellT, unsigned long int>
658  || std::is_same_v<CellT, unsigned long long int>) {
659  return readNumber<CellT>(cell).value_or(0);
660  } else {
661  // Cell* classes
662  return CellT(cell, hint);
663  }
664 }
