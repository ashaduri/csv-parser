73std::string cleanString(std::string_view view)
74{
75 std::string s;
76 s.reserve(view.size());
77 for (std::size_t pos = 0; pos < view.size(); ++pos) {
78 char c = view[pos];
79 s += c;
80 if (c == '\"' && (pos + 1) < view.size() && view[pos + 1] == '\"') {
81 ++pos;
82 }
83 }
84 return s;
85}

89constexpr std::size_t getCleanStringSize(std::string_view uncollapsed_view)
90{
91 // Count the number of quotes in the string. Since quotes are doubled,
92 // subtract half.
93 // std::count is not constexpr in C++17, so we have to do it manually.
94 std::size_t quote_count = 0;
95 for (char c : uncollapsed_view) {
96 if (c == '\"') {
97 ++quote_count;
98 }
99 }
100 return uncollapsed_view.size() - (quote_count / 2);
101}

105template<typename Number>
106std::optional<Number> readNumber(std::string_view cell)
107{
108 // Trim right whitespace (left whitespace is ignored by stod()).
109 std::size_t size = cell.size();
110 if (auto end_pos = cell.find_last_not_of(" \t"); end_pos != std::string_view::npos) {
111 size = end_pos + 1;
112 }
113 std::string s(cell.data(), size);
114
115 // Convert to lowercase (needed for Matlab-produced CSV files)
116 std::transform(s.begin(), s.end(), s.begin(), [](unsigned char c) {
117 return static_cast<char>(std::tolower(c));
118 });
119
120 std::optional<Number> numeric_value;
121 // As of 2020, from_chars() is broken for floats/doubles in most compilers, so we'll have to do with stod()/... for
122 // now, even if it means using current locale instead of C locale.
123 // While calling std::strtod() could be potentially faster, it also means we have to deal with some
124 // platform-specific errno and other peculiarities. std::stod() wraps that nicely.
125 try {
126 std::size_t num_processed = 0;
127 Number parsed_value;
128
129 // We have to use a 0-terminated string in std::sto*() functions.
130 if constexpr(std::is_same_v<Number, float>) {
131 parsed_value = std::stof(s, &num_processed);
132 } else if constexpr(std::is_same_v<Number, double>) {
133 parsed_value = std::stod(s, &num_processed);
134 } else if constexpr(std::is_same_v<Number, long double>) {
135 parsed_value = std::stold(s, &num_processed);
136
137 } else if constexpr(std::is_same_v<Number, int>) {
138 parsed_value = std::stoi(s, &num_processed);
139 } else if constexpr(std::is_same_v<Number, long int>) {
140 parsed_value = std::stol(s, &num_processed);
141 } else if constexpr(std::is_same_v<Number, long long int>) {
142 parsed_value = std::stoll(s, &num_processed);
143
144 } else if constexpr(std::is_same_v<Number, unsigned long int>) {
145 parsed_value = std::stoul(s, &num_processed);
146 } else if constexpr(std::is_same_v<Number, unsigned long long int>) {
147 parsed_value = std::stoull(s, &num_processed);
148
149 } else {
150 static_assert(internal::always_false<Number>::value, "Invalid Number type in Csv::readNumber()");
151 }
152
153 if (num_processed == s.size()) {
154 numeric_value = parsed_value;
155 }
156 } catch (...) {
157 // nothing
158 }
159 return numeric_value;
160}
